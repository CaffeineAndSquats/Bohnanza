lng FLOAT,
organic BOOLEAN,
fielda INT,
fieldb INT,
warehouse BOOLEAN);

 id INT PRIMARY KEY AUTO_INCREMENT,

volume INT,
date TIMESTAMP

 id INT PRIMARY KEY AUTO_INCREMENT,

lng FLOAT,

date TIMESTAMP,
